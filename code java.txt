private static final int TIMEOUT = 2000;
private static final int DATA_RATE = 9600;

    public EP_JAVA_Frame() {
        initComponents();
        iniciarConexion();
    }
    
    public void iniciarConexion(){
        CommPortIdentifier puertoID = null;
        Enumeration puertoEnum = CommPortIdentifier.getPortIdentifiers();
        
        while (puertoEnum.hasMoreElements()){
            CommPortIdentifier actualPuertoID = (CommPortIdentifier) puertoEnum.nextElement();
            if (PUERTO.equals(actualPuertoID.getName())){
                puertoID = actualPuertoID;
                break;
            }
        }
        
        if (puertoID == null){
            mostrarError("No se puede conectar al puerto");
            System.exit(ERROR);
        }
        
        try{
            serialPort = (SerialPort) puertoID.open(this.getClass().getName(), TIMEOUT);
            serialPort.setSerialPortParams(DATA_RATE, SerialPort.DATABITS_8, SerialPort.STOPBITS_2, SerialPort.PARITY_NONE);
            output = serialPort.getOutputStream();
        } catch (Exception e){
            mostrarError(e.getMessage());
            System.exit(ERROR);
        }
        
        try {
            input = serialPort.getInputStream();
        } catch (IOException ex) {
            Logger.getLogger(EP_JAVA_Frame.class.getName()).log(Level.SEVERE, null, ex);
        }
        
        try {
            serialPort.addEventListener(this);
            serialPort.notifyOnDataAvailable(true);
        } catch (TooManyListenersException ex) {
            Logger.getLogger(EP_JAVA_Frame.class.getName()).log(Level.SEVERE, null, ex);
        }
        
    }
    
    
    @Override
    public void serialEvent(SerialPortEvent spe) {
        if (spe.getEventType() == SerialPortEvent.DATA_AVAILABLE){
            byte[] readBuffer = new byte[20];
            
            try{
                int numBytes = 0;
                while (input.available() > 0){
                    numBytes = input.read(readBuffer);
                }
                
                jTextArea2.append(new String(readBuffer, 0, numBytes, "us-ascii"));
                jTextArea2.setCaretPosition(jTextArea1.getDocument().getLength());
            }catch (IOException e){
                System.out.println(e);
            }
        }
    }
    
    
    public void enviarDatos(String datos){
        try{
            output.write(datos.getBytes());
        } catch (Exception e){
            mostrarError("ERROR");
            System.exit(ERROR);
        }
    }
    
    public void mostrarError(String mensaje){
        JOptionPane.showMessageDialog(this, mensaje, "ERROR", JOptionPane.ERROR_MESSAGE);
    }
