void rutina_java(){

  base_mem = base_i;
  base_i_mem = base_destino;

  base_destino = base_i_mem;
  while (base_i < base_destino){
    servo1.write(base_i);
    base_i = base_i + 5;
    delay(100);
  }

  base_i = base_mem;
  while (base_destino > base_i){
    servo1.write(srv_base);
    base_destino = base_destino - 5;
    delay(100);
  }
}

void setup() {
  // Iniciamos el monitor serie para mostrar el resultado
  Serial.begin(9600);
  // Iniciamos el servo para que empiece a trabajar con el pin 9
  servo1.attach(9);
  servo2.attach(8);
  servo3.attach(7);
  servo4.attach(2);
  servo5.attach(4);  

}

void postura_inicial(){

  servo4.write(srv_muneca);
  servo5.write(srv_pinza);
  delay(1000);
}

void seleccion_servo(){
    if (select_svr < 1){
      if (Serial.available()>0){
        input=Serial.read();
        if (input =='0'){
          compu = 0;
          select_svr = 0;
        }
        if (input =='1'){
          select_svr = 1;
          Serial.println("servo 1");
        }
        if (input =='2'){
          select_svr = 2;
          Serial.println("servo 2");
        }
        if (input =='3'){
          select_svr = 3;
          Serial.println("servo 3");
        }                        
        if (input =='4'){
          select_svr = 4;
          Serial.println("servo 4");
        }
        if (input =='5'){
          select_svr = 5;
          Serial.println("servo 5");
        }
        if (input == '6'){
          select_svr = 6;
          Serial.println("rutina java");
          rutina_java();          
        }        
      }    
    } else { mover_servo(); }
}

void mover_servo(){
     if (Serial.available()>0){
        input=Serial.read();
        if (input == '0'){
          compu = 0;
          select_svr = 0;
        }
        
        if (select_svr == 1) {
          if (input=='1'){
            srv_base = srv_base + 5;
            servo1.write(srv_base);
            Serial.println("servo 1 + 5");
          }          
          if (input=='2'){
            srv_base = srv_base - 5;
            servo1.write(srv_base);
            Serial.println("servo 1 - 5");
          }       
        }
